311 template <int step = 1>
312 string &operator+=(const string &rhs) {
313 if(!rhs)
314 return *this;
315 make_unique();
316 if(!*this || m_ln + rhs.m_ln >= m_sz - 1) {
317 const unsigned m = ((m_ln + rhs.m_ln + (step - 1)) / step);
318 unsigned req = m * step;
319 reserve(req);
320 }
321 memcpy(m_buf + m_ln, rhs.m_buf, (size_t)rhs.m_ln + 1);
322 m_ln = m_ln + rhs.m_ln;
323 return *this;
324 }
