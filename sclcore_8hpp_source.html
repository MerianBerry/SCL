368 template <int step = 1>
369 scl::string &operator+=(const scl::string &rhs) {
370 if(!rhs)
371 return *this;
372 make_unique();
373 if(!*this || m_ln + rhs.m_ln >= m_sz - 1) {
374 const unsigned m = ((m_ln + rhs.m_ln + (step - 1)) / step);
375 unsigned req = m * step;
376 reserve(req);
377 }
378 memcpy(m_buf + m_ln, rhs.m_buf, (size_t)rhs.m_ln + 1);
379 m_ln = m_ln + rhs.m_ln;
380 return *this;
381 }
