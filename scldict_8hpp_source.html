67 hnode *n = gnodebase(hash);
68 while(n && n->m_hash != hash && n->m_next)

84 hnode *nodenext(hnode *node) const {
85 if(!node)
86 return first();
87 // If there is another node in the chain
88 if(node->m_next)
89 return node->m_next;
90 // Continue on from the next node in the htab array
91 for(unsigned i = nodei(node->m_hash) + 1; i < 1u << m_hsz; i++)
92 if(m_ht[i])
93 return m_ht[i];
94 // No next node could be found
95 return nullptr;
96 }

98 void mutate(bool free) override {
99 dictionary fun;
100 for(hnode *n = nodenext(nullptr); n;) {
101 fun.set(n->m_key, n->m_data);
102 n = nodenext(n);
103 }
104 if(free)
105 _clear();
106 *this = fun;
107 }

109 void put(hnode *node) {
110 node->m_next = nullptr;
111 make_unique();
112 hnode *n = gnodebase(node->m_hash);
113 while(n && n->m_next && n->m_hash != node->m_hash)
114 n = n->m_next;
115 m_hnum++;
116 // New node in the array
117 if(!n)
118 m_ht[nodei(node->m_hash)] = node;
119 // Replace existing nodes data
120 else if(n->m_hash == node->m_hash) {
121 m_hnum--;
122 n->m_data = std::move(node->m_data);
123 delete node;
124 } else
125 // Chain node
126 n->m_next = node;
127 }

129 void rehash(hnode **oh, unsigned ohsz) {
130 for(unsigned i = 0; i < ((unsigned)1 << ohsz); i++) {
131 hnode *n = oh[i];
132 for(; n;) {
133 hnode *m_next = n->m_next;
134 put(n);
135 n = m_next;
136 }
137 oh[i] = nullptr;
138 }
139 }

141 void optimize() {
142 hnode **oh = m_ht;
143 unsigned char ohsz = m_hsz;
144 m_hsz = optimal();
145 unsigned s = (1 << m_hsz);
146 m_ht = new hnode *[s];
147 if(!m_ht)
148 throw "out of memory";
149 memset(m_ht, 0, sizeof(hnode *) * s);
150 m_hnum = 0;
151 if(oh)
152 rehash(oh, ohsz), free((void *)oh);
153 }

155 void _clear() {
156 hnode *h = nodenext(nullptr);
157 for(; h;) {
158 hnode *next = nodenext(h);
159 delete h;
160 if(next)
161 h = next;
162 else
163 h = 0;
164 }
165 delete[] m_ht;
166 m_ht = nullptr;
167 m_hsz = 0;
168 m_hnum = 0;
169 }

274 void remove(const K &key) {
275 unsigned hash = ghash(key);
276 hnode *n = gnodebase(hash);
277 while(n && n->m_next && n->m_hash != hash)
278 n = n->m_next;
279dict_remove_begin:
280 if(!n || n->m_hash != hash)
281 return;
282 if(n && n->m_hash == hash) {
283 if(m_ht[nodei(hash)] == n)
284 m_ht[nodei(hash)] = n->m_next;
285 delete n;
286 m_hnum--;
287 if(!isoptimal())
288 optimize();
289 } else if(n->m_next && n->m_next->m_hash == hash) {
290 hnode *prev = n;
291 n = n->m_next;
292 prev->m_next = n->m_next;
293 goto dict_remove_begin;
294 }
295 }
