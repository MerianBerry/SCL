39 static unsigned ghash (K const &key) {
40 return Hfunc::hash (key);

43 bool isoptimal() const {
44 bool nstrained = m_hnum > (1 << (m_hsz - 1));
45 bool nbig = m_hnum > (m_hsz >> 3) && m_hsz > SCL_DICT_MIN;
46 return !nstrained && !nbig && m_hsz >= SCL_DICT_MIN;
47 }
48
49 uchar optimal() const {
50 return std::max (log2i (m_hnum) + 2, SCL_DICT_MIN);
51 }
52
53 unsigned nodei (unsigned hash) const {
54 return hash % (1 << m_hsz);
55 }
56
57 hnode *gnodebase (unsigned hash) const {
58 if (!m_ht || !hash)
59 return nullptr;
60 return m_ht[nodei (hash)];
61 }

63 hnode *gnodefull (unsigned hash) const {
64 hnode *n = gnodebase (hash);
65 while (n && n->m_hash != hash && n->m_next)
66 n = n->m_next;
67 if (n && n->m_hash == hash)
68 return n;
69 return nullptr;
70 }

72 hnode *first() const {
73 for (unsigned i = 0; i < 1 << m_hsz; i++) {
74 hnode *n = m_ht[i];
75 if (m_ht[i])
76 return m_ht[i];
77 }
78 return nullptr;
79 }

81 hnode *nodenext (hnode *node) const {
82 if (!node)
83 return first();
84 // If there is another node in the chain
85 if (node->m_next)
86 return node->m_next;
87 // Continue on from the next node in the htab array
88 for (unsigned i = nodei (node->m_hash) + 1; i < 1 << m_hsz; i++)
89 if (m_ht[i])
90 return m_ht[i];
91 // No next node could be found
92 return nullptr;
93 }

95 void mutate (bool free) override {
96 dictionary fun;
97 for (hnode *n = nodenext (nullptr); n;) {
98 fun.set (n->m_key, n->m_data);
99 n = nodenext (n);
100 }
101 if (free)
102 _clear();
103 *this = fun;
104 }
105
106 void put (hnode *node) {
107 node->m_next = nullptr;
108 make_unique();
109 hnode *n = gnodebase (node->m_hash);
110 while (n && n->m_next && n->m_hash != node->m_hash)
111 n = n->m_next;
112 m_hnum++;
113 // New node in the array
114 if (!n)
115 m_ht[nodei (node->m_hash)] = node;
116 // Replace existing nodes data
117 else if (n->m_hash == node->m_hash) {
118 n->m_data = node->m_data;
119 delete node;
120 } else
121 // Chain node
122 n->m_next = node;
123 }
124
125 void rehash (hnode **oh, unsigned ohsz) {
126 for (unsigned i = 0; i < ((unsigned)1 << ohsz); i++) {
127 hnode *n = oh[i];
128 for (; n;) {
129 hnode *m_next = n->m_next;
130 put (n);
131 n = m_next;
132 }
133 oh[i] = nullptr;
134 }
135 }
136
137 void optimize() {
138 hnode **oh = m_ht;
139 unsigned char ohsz = m_hsz;
140 m_hsz = optimal();
141 unsigned s = (1 << m_hsz);
142 m_ht = new hnode *[s];
143 if (!m_ht)
144 throw "out of memory";
145 memset (m_ht, 0, sizeof (hnode *) * s);
146 m_hnum = 0;
147 if (oh)
148 rehash (oh, ohsz), free ((void *)oh);
149 }
150
151 void _clear() {
152 hnode *h = nodenext (nullptr);
153 for (; h;) {
154 hnode *next = nodenext (h);
155 delete h;
156 if (next)
157 h = next;
158 else
159 h = 0;
160 }
161 delete[] m_ht;
162 m_ht = nullptr;
163 m_hsz = 0;
164 m_hnum = 0;
165 }

170 m_hnum = 0;

172 }

176 }

178 ~dictionary() override {
179 if (deref() && m_ht)

181 }

187 void clear() {
188 make_unique (false);
189 _clear();
190 }

198 void set (K const &key, T const &v) {
199 // Make this unique
200 make_unique();
201 if (!isoptimal())
202 optimize();
203 hnode *node = new hnode();
204 if (!node)
205 throw "out of memory";
206 node->m_key = key;
207 node->m_data = v;
208 node->m_hash = ghash (key);
209 node->m_next = nullptr;
210 put (node);
211 }
212
219 void remove (K const &key) {
220 unsigned hash = ghash (key);
221 hnode *n = gnodebase (hash);
222 while (n && n->m_next && n->m_next->m_hash != hash)
223 n = n->m_next;
224 if (n->m_next && n->m_next->m_hash == hash) {
225 hnode *node = n->m_next;
226 n->m_next = node->m_next;
227 delete node;
228 return;
229 } else if (n->m_hash == hash) {
230 unsigned i = nodei (hash);
231 m_ht[i] = n->m_next;
232 delete n;
233 }
234 }

236 htab_iterator begin() {
237 return htab_iterator (nodenext (nullptr), *this);
238 }
239
240 htab_iterator end() {
241 return htab_iterator (nullptr, *this);
242 }
243
251 htab_iterator get (K const &key) {
252 unsigned hash = ghash (key);
253 hnode *n = gnodefull (hash);
254 if (n)
255 return htab_iterator (n, *this);
256 return end();
257 }
258
266 htab_iterator operator[] (K const &key) {
267 htab_iterator i = get (key);
268 if (i != end())
269 return i;
270 return htab_iterator (key, *this);
271 }
