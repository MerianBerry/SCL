317 template <int step = 1>
318 scl::string& operator+=(const scl::string& rhs) {
319 if(!rhs)
320 return *this;
321 make_unique();
322 if(!*this || m_ln + rhs.m_ln >= m_sz - 1) {
323 const unsigned m = ((m_ln + rhs.m_ln + (step - 1)) / step);
324 unsigned req = m * step;
325 reserve(req);
326 }
327 memcpy(m_buf + m_ln, rhs.m_buf, (size_t)rhs.m_ln + 1);
328 m_ln = m_ln + rhs.m_ln;
329 return *this;
330 }
